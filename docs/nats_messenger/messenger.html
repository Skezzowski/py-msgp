221    async def response(
222        self, subject: str, service_fun: Callable[[bytes, dict], tuple[bytes, dict]]
223    ):
224        """
225        Subscribes to the `subject` topic, and calls the `service_fun` call-back function
226        with the inbound messages, then respond with the return value of the `service` function.
227
228        Args:
229          subject: Subject that the service as a subscriber will observe.
230          service_fun: a Callable function. Its return value will be the response.
231        """
232
233        async def nats_callback(msg):
234            self.logger.debug(f"Call service function with '{msg}'")
235            service_response, service_response_headers = await service_fun(
236                msg.data, msg.headers
237            )
238            self.logger.debug(
239                f"Respond with payload: '{service_response}', headers: {service_response_headers}"
240            )
241            if service_response is None:
242                service_response = b""
243            await self.nats_conn.publish(
244                subject=msg.reply,
245                payload=service_response,
246                headers=service_response_headers,
247            )
248
249        subscription = await self.nats_conn.subscribe(subject=subject, cb=nats_callback)
250        subs = Subscriber(self.nats_conn, subscription)
251        self.logger.debug(f"Subscribed to {subject} via subscriber: {subs}")
252        return subs

318    async def subscribe_durable_with_ack(
319        self, subject: str, callback: Callable[[bytes, dict], None]
320    ):
321        """
322        Subscribes to the durable `subject`, and call `callback` with the received content.
323        The callback must return the content, or None if it failed.
324
325        Args:
326          subject: Subject that the subscriber will observe.
327          callback: a Callable function, that the subscriber will call.
328        """
329
330        async def js_callback(msg):
331            self.logger.debug(f"Subscription callback function is called with '{msg}'")
332            acknowledge = await callback(msg.data, msg.headers)
333            if acknowledge is None:
334                await msg.nak()
335                self.logger.debug(f"message {msg}, is NOT acknowledged")
336            else:
337                await msg.ack()
338                self.logger.debug(f"message {msg}, is acknowledged")
339
340        subscription = await self.js_conn.subscribe(
341            subject=subject,
342            cb=js_callback,
343            manual_ack=True,
344        )
345        subs = Subscriber(self.js_conn, subscription)
346        self.logger.debug(f"Subscribed to {subject} via subscriber: {subs}")
347        return subs
